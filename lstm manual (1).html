""" 
For new input x2
we have  h1,c1 updated from previous lstm

"""

neti2= w[0][0]*x2+w[0][1]*h1+w[0][2]
i2=1./(1+np.exp(-neti2)) 


netc2i= w[1][0]*x2+w[1][1]*h1+w[1][2]
c2i=np.tanh(netc2i)
 
netf2=w[2][0]*x1+w[2][1]*h1+w[2][2]
f2=1./(1+np.exp(-netf2))

 
    
c2 = i2*c2i + f2*c1
 
neto2= w[3][0]*x1+w[3][1]*h1+w[3][2]
o2=1./(1+np.exp(-neto2)) 
 
h2= o2*np.tanh(c2)
print(" Updated c2,h2 are " , c2,h2)
